Oura's API blocks direct browser requests, so you need a CORS proxy. Enter your proxy URL and Personal Access Token below — both stored only in your browser.

Should forward requests to https://api.ouraring.com · see Deno example ↓
Deno Deploy script:
Deno.serve(async (req) => {
  if (req.method === "OPTIONS") return new Response(null, {
    headers: { "Access-Control-Allow-Origin": "*",
      "Access-Control-Allow-Headers": "Authorization, Content-Type" }});
  const url = new URL(req.url);
  const resp = await fetch("https://api.ouraring.com" + url.pathname + url.search,
    { headers: { "Authorization": req.headers.get("Authorization") || "" }});
  return new Response(await resp.text(), { status: resp.status,
    headers: { "Content-Type": "application/json", "Access-Control-Allow-Origin": "*",
      "Access-Control-Allow-Headers": "Authorization" }});
});
